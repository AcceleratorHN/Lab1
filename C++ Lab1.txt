#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>

using namespace std;

vector<pair<int, int>> factorize(int x) {
    vector<pair<int, int>> factors;
    for (int i = 2; i * i <= x; ++i) {
        if (x % i == 0) {
            int cnt = 0;
            while (x % i == 0) {
                x /= i;
                cnt++;
            }
            factors.emplace_back(i, cnt);
        }
    }
    if (x > 1) {
        factors.emplace_back(x, 1);
    }
    return factors;
}

int main() {
    int A, B;
    cin >> A >> B;

    auto a_factors = factorize(A);
    int n = 1;

    for (auto [p, exp_a] : a_factors) {
        if (B % p != 0) {
            cout << -1 << endl;
            return 0;
        }
        int exp_b = 0;
        int temp = B;
        while (temp % p == 0) {
            temp /= p;
            exp_b++;
        }
        // Находим минимальное n, такое что n * exp_b >= exp_a
        int required_n = (exp_a + exp_b - 1) / exp_b; // Округление вверх
        n = max(n, required_n);
    }
    cout << "Семенов Дмитрий Викторович 020303-АИСа-о24" << endl;
    cout << n << endl;
    return 0;
}
